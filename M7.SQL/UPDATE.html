Câu lệnh UPDATE được sử dụng để cập nhật dữ liệu cho 1 hay nhiều bản ghi
    CÚ pháp :
    UPDATE table_name
SET cột 1 = giá trị mới của cột 1, cột 2 = giá trị mới cột 2, ...
WHERE điều kiện;

    Ví dụ cập nhật bản ghi có id = 1 trong bảng customers set tên là t3h và họ là ha noi
    UPDATE customers SET first_name = 't3h', last_name= 'ha noi' WHERE id = 11
