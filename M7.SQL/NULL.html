giá trị NULL trong SQL
        NULL có nghĩa là không có giá trị
        nhưng NULL lại khác ''
        NULL khác rỗng

        1 Cú pháp để kiểm tra xem cột đó có giá trị là NULL
        SELECT column_names
FROM table_name
WHERE column_name IS NULL;

        2 Cú pháp để kiểm tra xem cột đó có giá trị không phải là NULL
        SELECT column_names
FROM table_name
WHERE column_name IS NOT NULL;
