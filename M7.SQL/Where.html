Mệnh đề WHERE
    được dùng để lọc kết quả từ câu truy vấn
    cứ pháp:
    SELECT cột 1, cột 2,...
    FROM tên bảng
    WHERE điều kiện;

    ví dụ 1:
    lấy ra tất cả các bản ghi có id = 5
    SELECT * FROM customers WHERE id = 5

    ví dụ 2:
    lấy ra tất cả các khách hàng đến từ Chicago
    SELECT * FROM customers WHERE city = 'Chicago'
    Lưu ý: là 1 số như id thì không cần có nháy bao nhưng chuỗi thì cần
    nháy đơn hoặc nháy kép để bao

    ví dụ 3:
    lấy ra tất cả các khách hàng có id lớn hơn 5
    SELECT * FROM customers WHERE id > 5

   Ví dụ 4 : lọc theo nhiều điều kiện
    AND ( đáp ứng tất cả các điều kiện )
    OR ( chỉ cần đáp ứng 1 trong các điều kiện )
    mệnh đề OR
    SELECT * FROM customers WHERE city = 'Chicago' OR city = 'Los Angelas'
    mệnh đề AND

    lất ra tất cả các khách hàng có nghề nghiệp là Purchasing Manager
    và đến từ Chicago
    SELECT id, job_title , last_name , first_name , city FROM customers WHERE job_title = 'Purchasing Manager' AND city = 'Chicago'

    Ví dụ 5:
    Mệnh đề AND

    lấy ra tất cả các khách hàng có nghề nghiệp là Purchasing Manager
    và đến từ Chicago
        SELECT id , job_title , last_name , first_name, city FROM customers WHERE job_title ='Purchasing Manager' AND city = 'Chicago'

    Ví dụ 6: từ khóa NOT có ý nghĩa phủ định
    lấy ra tất cả các khách hàng àm không đến từ 'Chicago'
    SELECT id, job_title , last_name , first_name , city FROM customers WHERE NOT city = 'Chicago'
